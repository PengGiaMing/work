import os
from os.path import join, getsize

for root, dirs, files in os.walk("c:\\"):
    fsum = 0
    for name in files:
        fname = join(root, name)
        fsize = getsize(fname)
        fsum += fsize
        print("%s => %d" % (fname, fsize))
    print("%s => %d" % (root, fsize))
