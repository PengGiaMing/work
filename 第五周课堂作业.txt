3.import datetime
def time(func):
    def wrapper(*args,**kw):
        star_time = datetime.datetime.now()
        func()
        over_time = datetime.datetime.now()
        print(f'需要时间{(over_time-star_time).total_seconds()}')
    return wrapper
@time
def demo():
    for i in range(10000):
        print(i)
demo()


4.def test(func):
    def wrapper(*args,**kw):
        try:
            func(*args,**kw)
        except ZeroDivisionError:
            print("除数不能为0")
        else:
            return func(*args,**kw)
    return wrapper

@test
def div(a,b):
    print(a/b)
div(1,0)



5.def now_time(fn):
    def wrapper():
        import datetime
        rs = fn()
        print("Done")
        print(datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S'))
        return rs
    return wrapper


@now_time
def func():
    print('start')
    for x in range(10):
        print(x)

func()
